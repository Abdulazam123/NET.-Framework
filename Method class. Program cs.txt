using System;

// Yeh class hai jo humne create ki
class Calculator
{
    // Yeh method 2 integers leta hai aur void hai (kuch return nahi karta)
    public void DoOperation(int num1, int num2)
    {
        // Pehle number par math operation (example: +10)
        int result = num1 + 10;

        // Result display karna (optional but useful)
        Console.WriteLine("Result after adding 10 to first number: " + result);

        // Doosra number screen par display karna
        Console.WriteLine("Second number is: " + num2);
    }
}

class Program
{
    static void Main(string[] args)
    {
        // Class ka object create kiya
        Calculator obj = new Calculator();

        // Method call simple tareeqay se (passing values)
        obj.DoOperation(5, 15);

        Console.WriteLine("------------------");

        // Method call using named parameters
        obj.DoOperation(num1: 20, num2: 30);

        // Program rukay takay output nazar aaye
        Console.ReadLine();
    }
}
